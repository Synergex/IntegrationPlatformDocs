Database Configuration

Example Configuration

 

 

{

  "Database": {

    "Server": "localhost",

    "Database": "KafkaReplication",

    "User": "KafkaReplicator",

    "Password": "p@ssw0rd",

    "AdminUser": "sa",

    "AdminPassword": "p@ssw0rd",

    "BatchSize": 10000,

    "BatchTimeoutSeconds": 0,

    "CompressionMode": "None",

    "RetryDelaySeconds": 30,

    "RetryMax": 10,
